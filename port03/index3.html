createPop({
    name: "#popup",

var isCookie = document.cookie.indexOf("popup=done");

if(isCookie == -1){
    $("#popup").show();
    $(".mask").show();
}else{
    $("#popup").hide();
    $(".mask").hide();
}

$("body").on("click", "#popup .close",function(e){
    e.preventDefault();
    removePop(this);        
});

function setCookie(cookieName, cookieVal, time){     
    var today = new Date();
    var date = today.getDate();
    today.setDate(date+time);     
    var duedate = today.toGMTString();
    document.cookie = cookieName+"="+cookieVal+"; path=/; expires="+duedate;
}

function createPop(opt){  
    var def_opt = {
        name: "#popup",
        data_url: "pop/pop.html",
        isMask: false,
        isCk: false,
        custom_css: undefined
    }
    var opt = Object.assign({}, def_opt, opt);

    if(opt.isMask){
        $("body").append(
            $("<div class='mask'>").css({
                width: '100%', height: '100%',  backgroundColor: 'rgba(0,0,0,0.2)',
                position: "fixed", top: 0, left: 0,  zIndex: 11, display:"none"
            }).fadeIn()
        )            
    }

    var result_css;

    var def_css = {
            position:'absolute', top: '50%', left:'50%', 
            transform: 'translate(-50%,-50%)', zIndex: 12, display:"none"
    }    

    if(opt.custom_css == undefined){
        result_css = def_css;
    }else{
        result_css = Object.assign({}, def_css, opt.custom_css);
    }        

    var id_name = opt.name.split("#")[1];      

    $("body").append(  
        $("<aside>").attr("id", id_name)
            .css(result_css)
            .append(
                $("<div class='content'>"),
                $("<div class='wrap'>")
                    .append(
                        $("<input type='checkbox' id='ck'>")
                        .css({cursor:"pointer",marginRight:10}),
                        $("<label for='ck'>").text("오늘 하루 보지 않기")
                        .css({cursor:"pointer"})
                    ),
                $("<a href='#'>").addClass("close").text("닫기")
            ).fadeIn()
    );
    
    if(!opt.isCk) $(opt.name).find(".wrap").hide();      

    $.ajax({
        url: opt.data_url,
        success : function(data){
            $(opt.name).find(".content").html(data);
        },
        error : function() {
            console.error()
        }
    })
}

function removePop(el){
    // el == this
    var isChecked = $(el).prev().find("input[type=checkbox]").is(":checked");
    var id_name = $(el).parent().attr("id");    

    if(isChecked) setCookie(id_name, "done", 1);

    $(el).parent().fadeOut(500, function(){
        $(this).remove();
    });
    
    $(".mask").fadeOut(500,function(){
        $(this).remove();
    })
    
}

    timer = setInterval(move ,20);

    e.preventDefault();

    if(isDone){
        isDone = false;
        next();
    }

$prev.on("click",function(e){
    e.preventDefault();

    if(isDone){
        isDone=false;
        prev(); 
    }               
});

function move(){
    if(i <= -wid*2){
        i= -wid;
        $list.children("li").first().appendTo($list);
    }else{
        i--;
    } 
    $list.css({marginLeft : i});
}

function next(){
    $list.animate({marginLeft : -wid*2},speed,function(){
        $list.children("li").first().appendTo($list);
        $list.css({marginLeft : -wid});
        i = -wid;
        isDone = true;                   
    });
}

function prev(){
    $list.animate({marginLeft : 0},speed,function(){
        $list.children("li").last().prependTo($list);
        $list.css({marginLeft : -wid});
        i= -wid;
        isDone = true; 
    });
}

            $scrollbtns.eq(i).find("a").addClass("on");

            $scrollboxs.eq(i).addClass("on");

    }        
}

WAVE ANIMATION

clip-path와 키프레임을 이용한 물결 애니메이션입니다.opacity, 각 wave의 animation-duration을 사용하여 공간감을 주었습니다.

MOUSE FOLLOWING ANIMATION

mousemove 이벤트로 cursor position을 구하고, 요소의 position에 이용한 애니메이션입니다. 부분 별로 마우스 위치값을 계산해 움직임의 속도를 다르게 주었습니다.

MASK ANIMATION

svg의 마스크인 clipPath를 이용하여 물결모양을 만들고, path에 css로 animation을 주어, 움직이는 물결모양으로 사라지는 애니메이션을 주었습니다.

RIPPLE ANIMATION

원모양의 여러 <div>의 scale에 css 로 animation을 주었고, 각 <div>마다 animation-del0ay값을 다르게 주어, 잔물결처럼 보이도록 하였습니다.

ball ANIMATION

각각의 ball에는 scale을 css animation으로 변화를 주면서, ball을 감싸는 wrap에는 translateX값을 주어, scale값이 작아지면서 다른 ball과 겹쳐 원근감이 느껴지도록, z-index값으로 다른 ball과 겹쳐질때 위로 오도록 해주었다.
